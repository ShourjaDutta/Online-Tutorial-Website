Structures in C Programming

Syntax for Structure

A structure in C is a user-defined data type that allows grouping variables of different data types under a single name. The syntax for defining a structure is as follows:

struct structure_name {
 data_type member1;
 data_type member2;
 ...
};

Example

Example of using a structure in C:

#include<stdio.h>
struct Point {
 int x;
 int y;
};
int main() {
 struct Point p1;
 p1.x = 10;
 p1.y = 20;
 printf("Coordinates: (%d, %d)", p1.x, p1.y);
 return 0;
}

Program Questions

1. Write a C program to store and display information about a student using a structure.
2. Create a program to find the area of a rectangle using a structure.

Program Solutions

Solutions for the above questions:

// Program 1:
#include<stdio.h>
struct Student {
 char name[50];
 int age;
 float marks;
};
int main() {
 struct Student s1;
 printf("Enter student details:\n");
 printf("Name: ");
 scanf("%s", s1.name);
 printf("Age: ");
 scanf("%d", &s1.age);
 printf("Marks: ");
 scanf("%f", &s1.marks);
 printf("Student Details:\nName: %s\nAge: %d\nMarks: %.2f", s1.name, s1.age, s1.marks);
 return 0;
}
// Program 2:
#include<stdio.h>
struct Rectangle {
 int length;
 int width;
};
int main() {
 struct Rectangle r1;
 printf("Enter dimensions of the rectangle:\n");
 printf("Length: ");
 scanf("%d", &r1.length);
 printf("Width: ");
 scanf("%d", &r1.width);
 int area = r1.length * r1.width;
 printf("Area of the rectangle: %d", area);
 return 0;
}
